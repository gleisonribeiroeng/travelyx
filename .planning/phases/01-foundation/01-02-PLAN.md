   - Remove: `import { provideAnimations } from '@angular/platform-browser/animations';`
   - Add: `import { provideAnimationsAsync } from '@angular/platform-browser/animations/async';`

2. **styles.scss** — Replace whatever the schematic generated with the M3 theme configuration:

```scss
@use '@angular/material' as mat;

// Apply M3 theme — generates all --mat-sys-* design tokens on html element
html {
  @include mat.theme((
    color: (
      theme-type: light,
      primary: mat.$azure-palette,
      tertiary: mat.$blue-palette,
    ),
    typography: Roboto,
    density: 0,
  ));
}

// Custom spacing tokens (supplement --mat-sys-* with project-specific values)
:root {
  --triply-spacing-xs: 4px;
  --triply-spacing-sm: 8px;
  --triply-spacing-md: 16px;
  --triply-spacing-lg: 24px;
  --triply-spacing-xl: 48px;

  --triply-border-radius-sm: 4px;
  --triply-border-radius-md: 8px;
  --triply-border-radius-lg: 16px;

  --triply-max-content-width: 1200px;
}

// Apply Material typography to body
body {
  font-family: Roboto, 'Helvetica Neue', sans-serif;
  margin: 0;
  padding: 0;
}
```

Note: Use `font-family` property directly instead of `font: var(--mat-sys-body-large)` shorthand, as the shorthand may not be available in all Material 3 versions. The Roboto font is loaded by the Material schematic (either via Google Fonts link in index.html or as a dependency).

3. **angular.json** — Verify the schematic did not add a prebuilt theme to the `styles` array. If it added something like `@angular/material/prebuilt-themes/...`, remove it since we are using the custom `mat.theme()` approach in `styles.scss`. The `styles` array should only contain `"src/styles.scss"`.

Create `triply/src/app/core/material.exports.ts` that re-exports commonly used Angular Material modules as a single importable array. This avoids every feature component importing 5-10 Material modules individually.

```typescript
// Core Material component re-exports for feature components.
// Import MATERIAL_IMPORTS in your component's `imports` array.
// Add new Material modules here as features require them.

import { MatButtonModule } from '@angular/material/button';
import { MatCardModule } from '@angular/material/card';
import { MatToolbarModule } from '@angular/material/toolbar';
import { MatIconModule } from '@angular/material/icon';
import { MatInputModule } from '@angular/material/input';
import { MatFormFieldModule } from '@angular/material/form-field';
import { MatDatepickerModule } from '@angular/material/datepicker';
import { MatNativeDateModule } from '@angular/material/core';
import { MatSelectModule } from '@angular/material/select';
import { MatChipsModule } from '@angular/material/chips';
import { MatDividerModule } from '@angular/material/divider';
import { MatProgressSpinnerModule } from '@angular/material/progress-spinner';
import { MatSnackBarModule } from '@angular/material/snack-bar';
import { MatTabsModule } from '@angular/material/tabs';

export const MATERIAL_IMPORTS = [
  MatButtonModule,
  MatCardModule,
  MatToolbarModule,
  MatIconModule,
  MatInputModule,
  MatFormFieldModule,
  MatDatepickerModule,
  MatNativeDateModule,
  MatSelectModule,
  MatChipsModule,
  MatDividerModule,
  MatProgressSpinnerModule,
  MatSnackBarModule,
  MatTabsModule,
] as const;
```

Verify the file compiles by running `npx ng build`.

Note: This is NOT a barrel file for cross-feature lazy loading (which the research warns against). This is an import-convenience array within `core/` — all modules listed are from `@angular/material` which is a single package already in the bundle. Feature components lazy-load their own code, not Material modules.
